async function loadTone() {

    if (window.Tone) return Promise.resolve();

    return new Promise((resolve, reject) => {

        const script = document.createElement('script');

        script.src = 'https://cdnjs.cloudflare.com/ajax/libs/tone/14.8.49/Tone.js';

        script.onload = () => resolve();

        script.onerror = () => reject(new Error("Failed to load Tone.js"));

        document.head.appendChild(script);

    });

}


async function initAudio() {

    await Tone.start();

    audioCtx = Tone.getContext().rawContext;


    masterFilter = new Tone.Filter(MATERIALS[activeMaterial].filter, "lowpass");

    masterReverb = new Tone.Reverb({ decay: 2.5, wet: 0.3 });

    limiter = new Tone.Limiter(-3);

    analyser = new Tone.Analyser('waveform', 2048); // Increased resolution for denser lines


    masterFilter.chain(masterReverb, limiter, analyser, Tone.Destination);

    await masterReverb.ready;


    analyserData = analyser.getValue();


    // Ambient Drone

    droneOsc = new Tone.Noise("pink");

    droneFilter = new Tone.AutoFilter({ frequency: 0.1, baseFrequency: 50, octaves: 2, filter: { type: "lowpass", Q: 1 } }).start();

    const droneGain = new Tone.Gain(0.02);

    droneOsc.chain(droneFilter, droneGain, masterFilter);

    droneOsc.start();

}


function playStoneSound(index, timeOffset = 0) {

    const freq = STONE_FREQS[index];

    const mat = MATERIALS[activeMaterial];

    const t = Tone.now() + timeOffset;


    const detune = (Math.random() - 0.5) * 10;

    const f = freq * Math.pow(2, detune / 1200);


    mat.ratios.forEach((ratio, i) => {

        const osc = new Tone.Oscillator({

            frequency: f * ratio,

            type: "sine"

        });

        const env = new Tone.Gain(0);


        osc.connect(env);

        env.connect(masterFilter);


        osc.start(t);

        env.gain.setValueAtTime(0, t);

        env.gain.linearRampToValueAtTime(mat.amps[i], t + 0.005);

        env.gain.exponentialRampToValueAtTime(0.001, t + mat.decays[i]);

        osc.stop(t + mat.decays[i] + 0.1);


        // Clean up Tone nodes to prevent memory leaks over time

        setTimeout(() => {

            try { osc.dispose(); env.dispose(); } catch(e) {}

        }, (timeOffset + mat.decays[i] + 1) * 1000);

    });

}


// --- VISUAL SYSTEM ---

let width, height;

const cam = { x: 0, y: 12, z: -25, fov: 600 };

let timeOffset = 0;

let lastTime = 0;


const ripples = [];

class Stone {

    constructor(index) {

        this.index = index;

        this.baseX = 0;

        this.baseZ = 0;

        this.y = 0;

        this.flash = 0;

        this.polyBase = [];

        this.polyTarget = [];

        

        const numPts = 5 + Math.floor(Math.random() * 3);

        for(let i=0; i<numPts; i++) {

            const angle = (i / numPts) * Math.PI * 2;

            const bRad = 1.2;

            const tRad = 0.8 + Math.random() * 1.0;

            this.polyBase.push({ a: angle, r: bRad });

            this.polyTarget.push({ a: angle + (Math.random()-0.5)*0.5, r: tRad });

        }

    }


    updatePos(w) {

        const isMobile = w < 768;

        if (isMobile) {

            const row = Math.floor(this.index / 5);

            const col = this.index % 5;

            this.baseX = (col - 2) * 5;

            this.baseZ = row === 0 ? 5 : 15;

        } else {

            this.baseX = (this.index - 4.5) * 5;

            this.baseZ = 10;

        }

    }


    hit() {

        this.flash = 1.0;

        ripples.push({

            x: this.baseX, z: this.baseZ, 

            time: 0, amp: 1.5, 

            speed: 12, freq: 2.0, falloff: 0.15

        });

    }


    draw(ctx, w, h, thm) {

        this.flash *= 0.92;

        

        // Morph shape

        const pts = [];

        for(let i=0; i<this.polyBase.length; i++) {

            const a = this.polyBase[i].a * (1-shapeFactor) + this.polyTarget[i].a * shapeFactor;

            const r = this.polyBase[i].r * (1-shapeFactor) + this.polyTarget[i].r * shapeFactor;

            pts.push({ x: this.baseX + Math.cos(a)*r, z: this.baseZ + Math.sin(a)*r });

        }


        const screenPts = pts.map(p => project(p.x, this.y, p.z, w, h));

        const center = project(this.baseX, this.y + 0.5, this.baseZ, w, h);


        ctx.beginPath();

        ctx.moveTo(screenPts[0].x, screenPts[0].y);

        for(let i=1; i<screenPts.length; i++) ctx.lineTo(screenPts[i].x, screenPts[i].y);

        ctx.closePath();


        const baseColor = hexToRgb(thm.accent);

        ctx.fillStyle = `rgba(${baseColor}, ${0.1 + this.flash * 0.6})`;

        ctx.fill();


        ctx.strokeStyle = thm.accent;

        ctx.lineWidth = 1 + this.flash * 2;

        ctx.shadowBlur = this.flash * 20;

        ctx.shadowColor = thm.accent;

        ctx.stroke();


        // Facets

        ctx.beginPath();

        for(let i=0; i<screenPts.length; i++) {

            ctx.moveTo(center.x, center.y);

            ctx.lineTo(screenPts[i].x, screenPts[i].y);

        }

        ctx.lineWidth = 0.5;

        ctx.stroke();

        ctx.shadowBlur = 0;


        // Store click target

        this.screenX = center.x;

        this.screenY = center.y;

    }

}


function project(x, y, z, w, h) {

    const scale = cam.fov / (z - cam.z);

    return {

        x: (x - cam.x) * scale + w/2,

        y: (y - cam.y) * scale + h/2,

        scale: scale

    };

}

    height = window.innerHeight;

    cvs.width = width;

    cvs.height = height;

    stones.forEach(s => s.updatePos(width));

}


function hexToRgb(hex) {

    const bigint = parseInt(hex.slice(1), 16);

    return `${(bigint >> 16) & 255}, ${(bigint >> 8) & 255}, ${bigint & 255}`;

}


// Draw Loop

function draw(time) {

    requestAnimationFrame(draw);

    const dt = (time - lastTime) / 1000 || 0;

    lastTime = time;

    timeOffset += dt;


    const thm = THEMES[activeTheme];


    ctx.globalCompositeOperation = 'source-over';

    ctx.fillStyle = thm.bg;

    ctx.fillRect(0, 0, width, height);


    ctx.globalCompositeOperation = 'lighter';


    // Update ripples

    for(let i = ripples.length-1; i>=0; i--) {

        ripples[i].time += dt;

        if(ripples[i].time > 3.0) ripples.splice(i, 1);

    }


    // Analyser Waveform mapping

    let waveData = null;

    if (analyser) {

        analyserData = analyser.getValue();

        waveData = analyserData;

    }


    // --- DRAW DENSE DYNAMIC WAVEFORM ---

    if (waveData) {

        ctx.save();

        ctx.translate(0, height / 2);

        ctx.globalCompositeOperation = 'lighter';


        const numSamples = waveData.length;

        const step = width / numSamples;


        // Pass 1: The thick, lower-opacity background haze (Gritty texture)

        ctx.beginPath();

        for(let i=0; i<numSamples; i++) {

            let x = i * step;

            let val = waveData[i];

            

            // Map screen X to approximate world X for ripple interaction

            let worldX = (x / width - 0.5) * 30; 

            let rOffset = 0;

            for(let rip of ripples) {

                const d = Math.abs(worldX - rip.x); 

                const phase = d * rip.freq - rip.time * rip.speed;

                if (phase < 0) {

                    const ripAmp = rip.amp * Math.exp(-d * rip.falloff) * Math.max(0, 1 - rip.time/3);

                    rOffset += Math.sin(phase) * ripAmp * 0.08; // Subtler ripple impact

                }

            }


            let combinedVal = val + rOffset;

            

            // Subtler jitter for less dynamic chaos

            let jitter = 1.0 + (Math.random() - 0.5) * 0.15;

            let yAmp = Math.abs(combinedVal) * height * 0.12 * jitter; // Reduced height

            

            ctx.moveTo(x, -yAmp);

            ctx.lineTo(x, yAmp);

        }

        ctx.lineWidth = 1;

        ctx.strokeStyle = `rgba(${hexToRgb(thm.sec)}, 0.03)`; // Extremely transparent secondary color

        ctx.stroke();


        // Pass 2: The sharper, brighter core peaks (Accent color)

        ctx.beginPath();

        for(let i=0; i<numSamples; i += 3) { // Skip samples for a "spiky" structured look

            let x = i * step;

            let val = waveData[i];

            

            let worldX = (x / width - 0.5) * 30; 

            let rOffset = 0;

            for(let rip of ripples) {

                const d = Math.abs(worldX - rip.x); 

                const phase = d * rip.freq - rip.time * rip.speed;

                if (phase < 0) {

                    const ripAmp = rip.amp * Math.exp(-d * rip.falloff) * Math.max(0, 1 - rip.time/3);

                    rOffset += Math.sin(phase) * ripAmp * 0.08; // Subtler ripple impact

                }

            }


            let combinedVal = val + rOffset;

            // Less random multiplier for a calmer peak structure

            let yAmp = Math.abs(combinedVal) * height * 0.08 * (0.9 + Math.random() * 0.2); 

            

            if (yAmp > 1) { // Only draw significant peaks

                ctx.moveTo(x, -yAmp);

                ctx.lineTo(x, yAmp);

            }

        }

        ctx.lineWidth = 1.5;

        ctx.strokeStyle = `rgba(${hexToRgb(thm.accent)}, 0.08)`; // Fainter accent color glow

        ctx.stroke();


        // Pass 3: The solid center origin line

        ctx.beginPath();

        ctx.moveTo(0, 0);

        ctx.lineTo(width, 0);

        ctx.lineWidth = 1;

        ctx.strokeStyle = `rgba(${hexToRgb(thm.sec)}, 0.15)`; // Fainter center line

        ctx.stroke();


        ctx.restore();

    }


    // Darken center behind the massive number to prioritize legibility

    const centerGrad = ctx.createRadialGradient(width/2, height/2, 0, width/2, height/2, height*0.35);

    centerGrad.addColorStop(0, `rgba(${hexToRgb(thm.bg)}, 0.85)`);

    centerGrad.addColorStop(1, `rgba(${hexToRgb(thm.bg)}, 0)`);

    ctx.globalCompositeOperation = 'source-over';

    ctx.fillStyle = centerGrad;

    ctx.fillRect(0, 0, width, height);


    // Draw Stones

    stones.forEach(s => s.draw(ctx, width, height, thm));

}

async function startAutoCountdown() {

    currentState = STATES.AUTO;

    btnBegin.parentElement.style.display = 'none';

    

    let count = 10;

    

    // Recursive loop for perfect 1s timing

    function playNext() {

        if (count > 0) {

            const stoneIdx = AUTO_SEQUENCE[10 - count];

            stones[stoneIdx].hit();

            playStoneSound(stoneIdx);

            updateNumberDisplay(count);

            count--;

            setTimeout(playNext, 1000);

        } else {

            setTimeout(enterComposeMode, 1500);

        }

    }

    

    updateNumberDisplay(10);

    playNext();

}

async function playSequenceLive() {

    if(isPlayingSequence) return;

    isPlayingSequence = true;

    

    const slots = document.querySelectorAll('.seq-slot');

    let step = 0;


    function playStep() {

        if(step >= 10) {

            isPlayingSequence = false;

            return;

        }


        slots.forEach(s => s.classList.remove('playing'));

        slots[step].classList.add('playing');

        

        const stoneIdx = sequence[step];

        if (stoneIdx !== null) {

            stones[stoneIdx].hit();

            playStoneSound(stoneIdx);

        }

        

        updateNumberDisplay(10 - step);

        step++;

        setTimeout(playStep, 1000);

    }

    

    playStep();

}


// --- EXPORT WAV (OFFLINE RENDERING) ---

async function recordAndExport() {

    if(isRecording || isPlayingSequence) return;

    isRecording = true;

    btnExport.style.display = 'none';

    

    // Play live visually while rendering in background

    playSequenceLive();


    const duration = 11.5; // 10s sequence + 1.5s tail

    const sampleRate = 44100;

    const offCtx = new OfflineAudioContext(2, sampleRate * duration, sampleRate);

    

    // Setup matching offline fx chain

    const offFilter = offCtx.createBiquadFilter();

    offFilter.type = "lowpass";

    offFilter.frequency.value = MATERIALS[activeMaterial].filter;


    // Use a simple convolver for offline reverb to guarantee rendering

    const offReverb = offCtx.createConvolver();

    offReverb.buffer = await generateImpulseResponse(offCtx, 2.5);


    // Wet/Dry mix for offline reverb

    const dryNode = offCtx.createGain(); dryNode.gain.value = 0.7;

    const wetNode = offCtx.createGain(); wetNode.gain.value = 0.3;


    offFilter.connect(dryNode);

    offFilter.connect(offReverb);

    offReverb.connect(wetNode);


    dryNode.connect(offCtx.destination);

    wetNode.connect(offCtx.destination);


    // Schedule sequence

    for(let i=0; i<10; i++) {

        if(sequence[i] !== null) {

            // Re-use tone voice logic but with offline context

            const freq = STONE_FREQS[sequence[i]];

            const mat = MATERIALS[activeMaterial];

            const t = i * 1.0;

            const f = freq * Math.pow(2, ((Math.random() - 0.5) * 10) / 1200);


            mat.ratios.forEach((ratio, j) => {

                const osc = offCtx.createOscillator();

                osc.type = "sine";

                osc.frequency.value = f * ratio;


                const gain = offCtx.createGain();

                gain.gain.value = 0;


                osc.connect(gain);

                gain.connect(offFilter);


                osc.start(t);

                gain.gain.setValueAtTime(0, t);

                gain.gain.linearRampToValueAtTime(mat.amps[j], t + 0.005);

                gain.gain.exponentialRampToValueAtTime(0.001, t + mat.decays[j]);

                osc.stop(t + mat.decays[j] + 0.1);

            });

        }

    }


    const renderedBuffer = await offCtx.startRendering();

    const wavBlob = audioBufferToWav(renderedBuffer);

    const url = URL.createObjectURL(wavBlob);

    

    btnExport.onclick = () => {

        const a = document.createElement('a');

        a.href = url;

        a.download = 'my-io-countdown.wav';

        a.click();

    };


    // Wait until live playback finishes before showing button

    setTimeout(() => {

        isRecording = false;

        btnExport.style.display = 'block';

    }, 10000);

}


// Generate basic synthetic IR for offline context

async function generateImpulseResponse(ctx, duration) {

    const length = ctx.sampleRate * duration;

    const impulse = ctx.createBuffer(2, length, ctx.sampleRate);

    for (let i = 0; i < 2; i++) {

        const channel = impulse.getChannelData(i);

        for (let j = 0; j < length; j++) {

            channel[j] = (Math.random() * 2 - 1) * Math.pow(1 - j / length, 3);

        }

    }

    return impulse;

}


// WAV Encoder

function audioBufferToWav(buffer) {

    const numOfChan = buffer.numberOfChannels;

    const length = buffer.length * numOfChan * 2 + 44;

    const bufferData = new ArrayBuffer(length);

    const view = new DataView(bufferData);

    const channels = [];

    let pos = 0, offset = 0;


    const setUint16 = data => { view.setUint16(pos, data, true); pos += 2; };

    const setUint32 = data => { view.setUint32(pos, data, true); pos += 4; };


    setUint32(0x46464952); // "RIFF"

    setUint32(length - 8); // file length - 8

    setUint32(0x45564157); // "WAVE"

    setUint32(0x20746d66); // "fmt " chunk

    setUint32(16); // length = 16

    setUint16(1); // PCM

    setUint16(numOfChan);

    setUint32(buffer.sampleRate);

    setUint32(buffer.sampleRate * 2 * numOfChan); // avg. bytes/sec

    setUint16(numOfChan * 2); // block-align

    setUint16(16); // 16-bit

    setUint32(0x61746164); // "data" - chunk

    setUint32(length - pos - 4); // chunk length


    for(let i=0; i<buffer.numberOfChannels; i++) channels.push(buffer.getChannelData(i));


    while(pos < length) {

        for(let i=0; i<numOfChan; i++) {

            let sample = Math.max(-1, Math.min(1, channels[i][offset]));

            sample = sample < 0 ? sample * 0x8000 : sample * 0x7FFF;

            view.setInt16(pos, sample, true);

            pos += 2;

        }

        offset++;

    }

    return new Blob([bufferData], {type: "audio/wav"});

}
